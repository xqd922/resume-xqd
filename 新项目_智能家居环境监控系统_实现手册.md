## 7. FreeRTOS 任务设计

### 7.1 任务列表

| 任务 | 优先级 | 栈大小 | 周期 | 职责 |
|------|--------|--------|------|------|
| `SensorTask` | 2 | 512 | 1s | 轮询采集所有传感器 |
| `LogicTask` | 3 | 512 | 500ms | 阈值判断 + 联动控制 |
| `OLEDTask` | 1 | 256 | 200ms | 刷新仪表盘 |
| `UartTxTask` | 2 | 512 | 事件 | 向 ESP32 上报 |
| `UartRxTask` | 3 | 512 | 事件 | 接收 ESP32 下发指令 |
| `EventTask` | 4（最高） | 256 | 事件 | 处理外部中断事件（人体感应） |

### 7.2 同步机制

| 资源 | 机制 | 保护目标 |
|------|------|----------|
| I2C 总线 | Mutex | 多任务共享 I2C |
| 传感器数据结构 | Mutex | 读写一致性 |
| ESP32 上行队列 | Queue | 异步消息传递 |
| 人体中断 | Binary Semaphore | ISR → Task |

### 7.4 典型任务：SensorTask

```c
/* app_sensor.h */
typedef struct {
    float    temperature;
    float    humidity;
    uint16_t light_lux;
    uint16_t mq2_raw;
    uint16_t mq135_raw;
    uint8_t  pir;           /* 人体感应 */
    uint32_t timestamp;
} SensorData_t;

extern SensorData_t g_sensor;

void Sensor_ReadAll(SensorData_t *s);
```

```c
/* app_sensor.c */
#include "app_sensor.h"
#include "app_task.h"
#include "aht20.h"
#include "bh1750.h"

SensorData_t g_sensor;

void vSensorTask(void *arg) {
    const TickType_t period = pdMS_TO_TICKS(1000);
    TickType_t last = xTaskGetTickCount();
    SensorData_t local;

    for (;;) {
        memset(&local, 0, sizeof(local));

        xSemaphoreTake(xI2CMutex, portMAX_DELAY);
        AHT20_ReadTempHumi(&local.temperature, &local.humidity);
        BH1750_ReadLux(&local.light_lux);
        xSemaphoreGive(xI2CMutex);

        local.mq2_raw   = ADC_ReadChannel(0);
        local.mq135_raw = ADC_ReadChannel(1);
        local.pir       = HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_0);
        local.timestamp = HAL_GetTick();

        xSemaphoreTake(xSensorMutex, portMAX_DELAY);
        memcpy(&g_sensor, &local, sizeof(local));
        xSemaphoreGive(xSensorMutex);

        /* 推送到上行队列 */
        char buf[128];
        int n = snprintf(buf, sizeof(buf),
            "{\"t\":%.1f,\"h\":%.1f,\"l\":%u,\"mq2\":%u,\"mq135\":%u,\"pir\":%u,\"ts\":%lu}",
            local.temperature, local.humidity, local.light_lux,
            local.mq2_raw, local.mq135_raw, local.pir, local.timestamp);
        xQueueSend(xMqttUpQueue, buf, 0);

        vTaskDelayUntil(&last, period);
    }
}
```

### 7.5 LogicTask（联动控制）

```c
typedef struct {
    uint8_t  auto_light;     /* 光暗自动开灯 */
    uint16_t light_threshold;
    uint8_t  gas_alarm;
    uint16_t mq2_threshold;
    uint32_t pir_last_trig;
    uint32_t pir_timeout_ms;
} LogicConfig_t;

LogicConfig_t g_cfg = {
    .auto_light      = 1,
    .light_threshold = 100,
    .gas_alarm       = 1,
    .mq2_threshold   = 2500,
    .pir_timeout_ms  = 30000,
};

void vLogicTask(void *arg) {
    const TickType_t period = pdMS_TO_TICKS(500);
    TickType_t last = xTaskGetTickCount();
    SensorData_t s;

    for (;;) {
        xSemaphoreTake(xSensorMutex, portMAX_DELAY);
        memcpy(&s, &g_sensor, sizeof(s));
        xSemaphoreGive(xSensorMutex);

        /* 人来+天黑 → 开灯 */
        if (g_cfg.auto_light && s.pir && s.light_lux < g_cfg.light_threshold) {
            Relay_Set(RELAY_LIGHT, 1);
            g_cfg.pir_last_trig = HAL_GetTick();
        }

        /* 无人 N 秒 → 自动关灯 */
        if (g_cfg.pir_last_trig && (HAL_GetTick() - g_cfg.pir_last_trig) > g_cfg.pir_timeout_ms) {
            Relay_Set(RELAY_LIGHT, 0);
            g_cfg.pir_last_trig = 0;
        }

        /* 气体超标 → 蜂鸣 + 排风 */
        if (g_cfg.gas_alarm && s.mq2_raw > g_cfg.mq2_threshold) {
            Buzzer_On();
            Relay_Set(RELAY_FAN, 1);
            SendEvent("gas_over", s.mq2_raw);
        }

        vTaskDelayUntil(&last, period);
    }
}
```

## 8. 传感器驱动实现

### 8.1 AHT20 温湿度

```c
/* aht20.c */
#include "aht20.h"

#define AHT20_ADDR  (0x38 << 1)

static HAL_StatusTypeDef AHT20_Trigger(I2C_HandleTypeDef *hi2c) {
    uint8_t cmd[3] = {0xAC, 0x33, 0x00};
    return HAL_I2C_Master_Transmit(hi2c, AHT20_ADDR, cmd, 3, 100);
}

HAL_StatusTypeDef AHT20_Init(I2C_HandleTypeDef *hi2c) {
    HAL_Delay(40);
    uint8_t cmd[3] = {0xBE, 0x08, 0x00};
    return HAL_I2C_Master_Transmit(hi2c, AHT20_ADDR, cmd, 3, 100);
}

HAL_StatusTypeDef AHT20_ReadTempHumi(I2C_HandleTypeDef *hi2c, float *t, float *h) {
    if (AHT20_Trigger(hi2c) != HAL_OK) return HAL_ERROR;
    HAL_Delay(80);

    uint8_t buf[7];
    if (HAL_I2C_Master_Receive(hi2c, AHT20_ADDR, buf, 7, 100) != HAL_OK) return HAL_ERROR;

    uint32_t hr = ((uint32_t)(buf[1]) << 12) | ((uint32_t)(buf[2]) << 4) | (buf[3] >> 4);
    uint32_t tr = (((uint32_t)(buf[3]) & 0x0F) << 16) | ((uint32_t)(buf[4]) << 8) | buf[5];

    *h = (float)hr * 100.0f / (1 << 20);
    *t = (float)tr * 200.0f / (1 << 20) - 50.0f;
    return HAL_OK;
}
```

### 8.2 BH1750 光照

```c
/* bh1750.c */
#include "bh1750.h"

#define BH1750_ADDR  (0x23 << 1)
#define CMD_CONT_H   0x10   /* Continuously H-Resolution Mode 1 lx/120ms */

HAL_StatusTypeDef BH1750_Init(I2C_HandleTypeDef *hi2c) {
    uint8_t cmd = CMD_CONT_H;
    return HAL_I2C_Master_Transmit(hi2c, BH1750_ADDR, &cmd, 1, 100);
}

HAL_StatusTypeDef BH1750_ReadLux(I2C_HandleTypeDef *hi2c, uint16_t *lux) {
    uint8_t buf[2];
    if (HAL_I2C_Master_Receive(hi2c, BH1750_ADDR, buf, 2, 100) != HAL_OK) return HAL_ERROR;
    uint16_t raw = (buf[0] << 8) | buf[1];
    *lux = raw / 1.2f;
    return HAL_OK;
}
```

### 8.3 MQ-2 模拟量

```c
/* mq_sensor.c */
uint16_t ADC_ReadChannel(uint8_t ch) {
    ADC_ChannelConfTypeDef sConfig = {0};
    sConfig.Channel      = (ch == 0) ? ADC_CHANNEL_0 : ADC_CHANNEL_1;
    sConfig.Rank         = 1;
    sConfig.SamplingTime = ADC_SAMPLETIME_71CYCLES_5;
    HAL_ADC_ConfigChannel(&hadc1, &sConfig);

    HAL_ADC_Start(&hadc1);
    HAL_ADC_PollForConversion(&hadc1, 10);
```

## 9. STM32 ↔ ESP32 通信协议

### 9.1 协议设计

```
帧格式：
┌────┬──────┬──────┬─────────────┬──────┬────┐
│0xAA│ CMD  │ LEN  │   PAYLOAD   │ CRC  │0x55│
│1 B │ 1 B  │ 2 B  │   N B       │ 2 B  │1 B │
└────┴──────┴──────┴─────────────┴──────┴────┘

CMD:
  0x01 DATA_REPORT       STM32 → ESP32 数据上报
  0x02 CMD_RELAY         ESP32 → STM32 继电器控制
  0x03 CMD_CONFIG        ESP32 → STM32 阈值配置
  0x04 EVENT_ALARM       STM32 → ESP32 告警事件
  0x05 HEARTBEAT         双向心跳
  0x06 ACK               应答
```

CRC16-CCITT 计算从 CMD 到 PAYLOAD 末尾。

### 9.2 编解码代码

```c
/* app_protocol.c */
#include "app_protocol.h"
#include "crc16.h"

#define FRAME_HEAD  0xAA
#define FRAME_TAIL  0x55

int Proto_Pack(uint8_t cmd, const uint8_t *payload, uint16_t len, uint8_t *out) {
    out[0] = FRAME_HEAD;
    out[1] = cmd;
    out[2] = len & 0xFF;
    out[3] = (len >> 8) & 0xFF;
    memcpy(&out[4], payload, len);
    uint16_t crc = CRC16_Calc(&out[1], 3 + len);
    out[4 + len]     = crc & 0xFF;
    out[4 + len + 1] = (crc >> 8) & 0xFF;
    out[4 + len + 2] = FRAME_TAIL;
    return 4 + len + 3;
}

int Proto_Unpack(const uint8_t *in, uint16_t total,
                 uint8_t *cmd, uint8_t *payload, uint16_t *len) {
    if (total < 7) return -1;
    if (in[0] != FRAME_HEAD || in[total-1] != FRAME_TAIL) return -2;
    *cmd = in[1];
    *len = in[2] | (in[3] << 8);
    if (total != 4 + *len + 3) return -3;
    uint16_t crc_calc = CRC16_Calc(&in[1], 3 + *len);
    uint16_t crc_recv = in[4 + *len] | (in[4 + *len + 1] << 8);
    if (crc_calc != crc_recv) return -4;
    memcpy(payload, &in[4], *len);
    return 0;
}
```

### 9.3 CRC16-CCITT

```c
uint16_t CRC16_Calc(const uint8_t *data, uint16_t len) {
    uint16_t crc = 0xFFFF;
    for (uint16_t i = 0; i < len; i++) {
        crc ^= (data[i] << 8);
        for (uint8_t j = 0; j < 8; j++) {
            crc = (crc & 0x8000) ? (crc << 1) ^ 0x1021 : (crc << 1);
        }
    }
    return crc;
}
```

| `home/{device_id}/status` | 设备 → 云 (Retained) | 1 | `{"online":true,"rssi":-52,"ver":"1.0.3"}` |

### 10.2 ESP32 MQTT 客户端

```cpp
/* esp32_mqtt.cpp */
#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

#define MQTT_HOST   "broker.emqx.io"
#define MQTT_PORT   1883
#define DEVICE_ID   "home001"

WiFiClient espClient;
PubSubClient mqtt(espClient);

void mqttCallback(char *topic, byte *payload, unsigned int length) {
    StaticJsonDocument<256> doc;
    deserializeJson(doc, payload, length);
    String t(topic);

    if (t.endsWith("/cmd")) {
        int relay = doc["relay"] | -1;
        bool state = doc["state"] | false;
        if (relay >= 0) {
            /* 打包帧转发 STM32 */
            uint8_t payload_buf[3] = { (uint8_t)relay, (uint8_t)state, 0 };
            uint8_t frame[16];
            int n = Proto_Pack(0x02, payload_buf, 2, frame);
            Serial2.write(frame, n);
        }
    } else if (t.endsWith("/config")) {
        /* 阈值配置下发 */
        int mq2 = doc["mq2_threshold"] | -1;
        if (mq2 > 0) {
            uint8_t payload_buf[2] = { (uint8_t)(mq2 & 0xFF), (uint8_t)(mq2 >> 8) };
            uint8_t frame[16];
            int n = Proto_Pack(0x03, payload_buf, 2, frame);
            Serial2.write(frame, n);
        }
    }
}

void mqttConnect() {
    String clientId = "ESP-" + String(DEVICE_ID);
    /* LWT：掉线自动发布 offline */
    String willTopic = "home/" + String(DEVICE_ID) + "/status";
    String willMsg   = "{\"online\":false}";
    while (!mqtt.connected()) {
        if (mqtt.connect(clientId.c_str(), NULL, NULL,
                         willTopic.c_str(), 1, true, willMsg.c_str())) {
            mqtt.subscribe(("home/" + String(DEVICE_ID) + "/cmd").c_str(), 1);
            mqtt.subscribe(("home/" + String(DEVICE_ID) + "/config").c_str(), 1);
            /* 上线通告 */
            mqtt.publish(willTopic.c_str(), "{\"online\":true}", true);
        } else {
            delay(2000);
        }
    }
}

void mqttPublishData(const String &json) {
    String topic = "home/" + String(DEVICE_ID) + "/data";
    mqtt.publish(topic.c_str(), json.c_str());
}
```

## 18. 面试应答模板

### Q1：这个项目最难的点是什么？怎么解决的？

> 最难的是**双 MCU 的可靠通信**。一开始用 ASCII 明文传输，在电机启停时 UART 经常收到毛刺数据导致 JSON 解析错误。后来重新设计了二进制帧协议：帧头帧尾做同步、CMD+LEN 明确结构、CRC16 校验整包、IDLE 中断做帧边界检测。为了防止接收缓冲错乱，用环形缓冲 + 状态机匹配帧头，即使中间丢字节也能快速恢复。最终在连续 24 小时压测下 0 误码。

### Q2：为什么用 FreeRTOS 而不是裸机？

> 裸机用超级循环 + 状态机也能做，但任务多了之后会难以维护：比如我这里有 5 个不同周期、不同优先级的任务，还要互斥访问 I2C 总线。用 FreeRTOS 后每个任务独立、时间基准清晰、Mutex/Queue 是成熟的同步机制，代码可读性和可维护性都显著提升。而且 RTOS 的抢占调度让告警任务（最高优先级）能立即抢占普通采集任务，响应更及时。

### Q3：I2C 多设备如何避免冲突？

> 三方面：**地址不同**（AHT20 是 0x38，BH1750 是 0x23，OLED 是 0x3C，互不冲突）；**上拉共用**（一组 4.7kΩ 上拉电阻足够）；**任务互斥**（FreeRTOS Mutex 保护，一次只允许一个任务占用总线）。实际跑起来 I2C 总线是单线程串行访问的。

### Q4：MQTT QoS 0/1/2 区别？你项目用哪个？

> - **QoS 0** 最多一次，发出就丢，不保证到达；
> - **QoS 1** 至少一次，可能重复；
> - **QoS 2** 恰好一次，握手 4 次最可靠但最慢。
>
> 我项目里：**传感数据**用 QoS 0（1 秒一次，丢一两条没关系）；**控制命令和告警**用 QoS 1（必须到，可以重复）；一般不用 QoS 2，开销太大。

### Q5：LWT（遗嘱消息）是什么？

> Last Will and Testament，是客户端连 Broker 时注册的一条"遗嘱"消息。当客户端非正常掉线（超过 Keep Alive 没心跳）时，Broker 会自动替它发布这条消息。我用 LWT 做设备离线感知：上线时发布 `{"online":true}` 为 retained，LWT 设为 `{"online":false}`，APP 订阅这个 Topic 就能立即知道设备状态。

### Q6：CRC 为什么用 CRC16-CCITT 而不是 CRC8？

> CRC8 检错能力有限，对长帧（几十字节）容易漏检；CRC16-CCITT 的多项式 0x1021 是经典选择，汉明距离好、抗突发错误强，嵌入式传输 100 字节以内几乎没有漏检。计算开销也不大，STM32 能跑到 MHz 级处理速度。

### Q7：怎么调试实时系统？

> - **串口打印** + 时间戳，把关键事件打出来。
> - **GPIO 翻转**标记任务进出，示波器看调度时序。
> - **FreeRTOS Runtime Stats** 看每个任务的 CPU 占用。
> - **STM32CubeMonitor** 实时监视变量。
> - **Tracealyzer**（付费）可视化任务切换。
> - 压测阶段用 Python 脚本模拟高频命令下发，观察系统稳定性。

### Q8：Deep Sleep 怎么实现平均 < 5mA？

> 核心是"大部分时间睡着，少量时间采集"。STM32 进 Stop 模式时 MCU 电流降到 10μA 级，外设关断。Wake 源：RTC 闹钟（每 10 分钟一次）+ PIR 外部中断（人来立即醒）。唤醒后快速采集 → 上报 → 再睡。计算下来采集占比 1%，空闲占比 99%，平均电流被稀释到单位 mA 级。

### Q9：OTA 失败如何保证不变砖？

> STM32 版：双分区（Active + Backup），新固件先写 Backup，CRC 校验通过再置位标志复位；Bootloader 检查标志后从 Backup 拷到 Active。中途掉电也无影响——下次启动还是旧固件，只是没升级成功。**永远保留一个能跑的分区**是 OTA 不变砖的核心。

### Q10：这个项目在简历上最能打的三个词？

> **分布式架构、FreeRTOS、云端联动**。三个词分别覆盖嵌入式系统设计能力、实时操作系统能力、IoT 全栈能力，基本能对上所有嵌入式/IoT 初级岗位的 JD 关键词。
