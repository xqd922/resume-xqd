cv2Impor tnumpy as np#加载图像image=cv2.imread("Cars2.tif",0)#加载为灰度图像#定义拉普拉斯核laplacian_kernel=np.array([[0,-1,0],[-1,5,-1],[0,-1,0]])#应用拉普拉斯核进行图像锐化sharpened_image=cv2.filter2D(image,-1,laplacian_kernel)#显示原始图像和锐化后的图像cv2.imshow("OriginalImage",image)cv2.imshow("car3
